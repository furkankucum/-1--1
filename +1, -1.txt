from distutils.command.build_scripts import first_line_re
import random




def randomNum():
    #uygun bir sayı olana kadar random üret ve döndür
    
    while True:
        sayı =random.randint(100,999)

        a = set()
        for i in str(sayı):
            a.add(int(i))
        if len(a) ==3:
            return(sayı)

        
def tahmin ():
    tahmin = int(input("> "))
    
def yorum ():
    tahmin = tahmin()
    sayı = randomNum()
    if tahmin == sayı:
        print("doğru")
        return

    tahmini=[]
    sayıi =[]
    for i in str(tahmin):
        tahmini.append(i)
    for i in str(sayı):
        sayıi.append(i)
    if tahmini[0] == sayıi[0]:
        print("+1")
        sayıi.remove[0]
    if tahmini[1] == sayıi[1]:
        print("+1")
        sayıi.remove[1]
    if tahmini[2] == sayıi[2]:
        print("+1")
        sayıi.remove[2]
    
    for i in tahmini:
        if i in sayıi:
            print("-1")
def main():
    print(yorum())
